# Py5Vector.copy

Create an identical copy of this Py5Vector instance.

v1 = py5.Py5Vector(1, 2, 3)

v2 = 10 * v1.copy

print(v1)
# Py5Vector3D([1., 2., 3.])
print(v2)
# Py5Vector3D([10., 20., 30.])
